from urllib.robotparser import RobotFileParser


class RobotsFilter:
    def __init__(self, target_domain, scheme="https"):
        self.target_domain = target_domain.lower()
        self.scheme = scheme
        self.robot_parser = RobotFileParser()
        self._load_robots_txt()

    def _load_robots_txt(self):
        """robots.txt 읽어오기"""
        robots_url = f"{self.scheme}://{self.target_domain}/robots.txt"
        self.robot_parser.set_url(robots_url)

        try:
            self.robot_parser.read()
            print(f"[robots.txt] 로드 완료: {robots_url}")
        except Exception:
            print(f"[robots.txt] 로드 실패 (무시하고 진행): {robots_url}")

    def is_allowed(self, url, user_agent="*"):
        """robots.txt 기준 크롤링 가능 여부"""
        try:
            allowed = self.robot_parser.can_fetch(user_agent, url)

            if not allowed:
                print(f"[robots.txt] 차단된 URL: {url}")

            return allowed

        except Exception as e:
            print(f"[robots.txt 검사 오류] {url} | {e}")
            return False
